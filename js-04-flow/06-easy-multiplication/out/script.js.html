/**
 * Iterates through numbers from 0 to 10 and logs the multiplication table of 9.
 * @example
 * // Logs: "9 * 0 = 0", "9 * 1 = 9", "9 * 2 = 18", ..., "9 * 10 = 90"
 */
for (let i = 0; i <= 10; i++) {
    let num = i * 9;
    console.log(`9 * ${i} = ${num}`);
}
